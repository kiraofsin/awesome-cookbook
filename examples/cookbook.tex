% This is a cookbook which shows off all the options of the awesome-cookbook
% template. The recipes are defined in the ./recipes folder.
%
% For more information see:
% https://github.com/Nooby4Ever/awesome-cookbook/README.md
%
%-------------------------------------------------------------------------------
%	CONFIGURATIONS
%
% 	NOTE: all commands except the \documentclass are optional!
%-------------------------------------------------------------------------------
% A4 paper size by default, use 'letterpaper' for US letter
\documentclass[11pt, a4paper]{awesome-cookbook}

% Specify your language
% Default: english
\setmainlanguage{english}

% Specify the directory which contain the fonts
% Default: fonts/
\setFontDir[fonts/]

% Specify the recipes folder if you use autogenerate
% Default: ./recipes/
\setRecipesDir[./recipes/]

% SPecify the images folder if you want to use \recipefigure
% (if you set this to "" you can specify full paths, instead of only the image name)
% Default: ./images/
\setImageDir[./images/]

% Specify the style for the category pages
% Default: style1
% Available: style1, style2
\setCategoryStyle[style1]

% Specify the default recipe header style, you can still specify 
% another style for a specific recipy by using \recipe[<style>]{<name>}. If
% the option is omitted this style specified with this command will be used.
% Default: style1
% Available: style1, style2, style3, style4
\setDefaultRecipeStyle[style1]

% Specify wether, by default, the recipes should always start on the
% left page. Note: this may cause blank pages. If you want to enable
% use: true (lower case).
% Default: false
\setDefaultRecipeStartLeft[false]

% Specify the icon size for the icons in the recipe.
% Default: 1cm
\setIconSize[1cm]

% Specify your cookbook name
% Default: The awesome
\cookbookName[The awesome]

% Specify the license for your cookbook
% Default: CC BY 4.0 international
% (you have to specify the whole text you want to show, may contain
% paragraphs)
%\setLicense{Your licence here}

% Author/Maintainer
% (note the { } brackets)
\author{Your Name}

% Specify the colors
% Defualt: awesome-red; awesome-darkgray; awesome-gray; awesome-text
% Available: awesome-emerald, awesome-skyblue, awesome-pink,
% 			 awesome-orange, awesome-concrete, awesome-darknight,
%			 awesome-purple, awesome-cyan, awesome-gray, awesome-red,
%			 awesome-darkgray, awesome-text
% You can also define your own colors an use them instead:
% \definecolor{mycolor1}{HTML}{414141}
% \definecolor{mycolor2}{rgb}{243,102,25}
\colorlet{awesome}{awesome-red}		% Highlights
\colorlet{base}{awesome-darkgray}		% Base color for the graphics
\colorlet{second}{awesome-gray}		% Secondary color
\colorlet{text}{awesome-text}			% Text color

% Specify the headers (for translations)
% Default: cookbook, Preface, Preparation, Ingredients, Alternatives, Notes,
% 			Energy list, {Nutrient, kcal, kj}, {min, servings, kj, €, source}
\nCookbook[cookbook]
\nPreface[Preface]
\nPreparation[Preparation]
\nIngredients[Ingredients]
\nAlternatives[Alternatives]
\nNotes[Notes]
\nEnergy[Energy list]
\nEnergyList{Nutrient}{kcal}{kj}		% header for energy list (appendix)
\nIcons{min}{servings}{kj}{€}{source}  % text under the icons, the 4th one is the
% currency symbole


%-------------------------------------------------------------------------------
%	DOCUMENT
%-------------------------------------------------------------------------------
\begin{document}

% Let the magic begin!
\autoGenerate

\end{document}
