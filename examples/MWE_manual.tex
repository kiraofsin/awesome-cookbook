% This is a MWE (minimal working example) for a manual cookbook,
% the recipes are defined in the recipe folder.
%
% For more information see:
% https://github.com/Nooby4Ever/awesome-cookbook/README.md
%
%-------------------------------------------------------------------------------
%	CONFIGURATION
%-------------------------------------------------------------------------------
% A4 paper size by default, use 'letterpaper' for US letter
\documentclass[11pt, a4paper]{awesome-cookbook}

%-------------------------------------------------------------------------------
%	DOCUMENT
%-------------------------------------------------------------------------------
\begin{document}
%
% The following code is equal to the code generated by \autoGenerate (autogenerate.lua)
%
% Generate the cover, title page, copyright page, TOC
\beginCookbook
\input{./recipes/preface.tex}
% Mark the tart of the recipes
\beginRecipes
\category{Starters}
\input{"./recipes/Starters/potato_rounds.tex"}
\input{"./recipes/Starters/rumaki.tex"}
\input{"./recipes/Starters/sausage_flowers.tex"}
\category{Main Courses}
\input{"./recipes/Main Courses/beef_stroganoff.tex"}
\input{"./recipes/Main Courses/chicken_tikka masala.tex"}
\category{Desserts}
\input{"./recipes/Desserts/peanut_blossoms.tex"}
\input{"./recipes/Desserts/zucchini_brownies.tex"}
% Mark the start of the appendices
\beginAppendices
\input{"./recipes/appendices/energylist.tex"}
% Note: the energy list itself will not be sorted!
\end{document}
